#include <stdio.h>

int main() {
    int n;

    // Get array size
    printf("Enter the number of elements: ");
    scanf("%d", &n);

    int arr[n], count[n];

    // Get array elements
    printf("Enter %d elements:\n", n);
    for (int i = 0; i < n; i++) {
        scanf("%d", &arr[i]);
        count[i] = -1; // Initialize count array
    }

    // Count occurrences
    for (int i = 0; i < n; i++) {
        int c = 1;
        for (int j = i + 1; j < n; j++) {
            if (arr[i] == arr[j]) {
                c++;
                count[j] = 0; // Mark as counted
            }
        }
        if (count[i] != 0) {
            count[i] = c;
        }
    }

    // Display results
    printf("Element | Count\n");
    for (int i = 0; i < n; i++) {
        if (count[i] != 0) {
            printf("   %d    |   %d\n", arr[i], count[i]);
        }
    }

    return 0;
}
