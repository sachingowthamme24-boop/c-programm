#include <stdio.h>

int main() {
    int r, c;
    printf("Enter number of rows and columns: ");
    scanf("%d %d", &r, &c);

    int A[r][c], B[r][c], C[r][c], SUM[r][c];

    printf("Enter elements of Matrix A:\n");
    for (int i = 0; i < r; i++)
        for (int j = 0; j < c; j++)
            scanf("%d", &A[i][j]);

    printf("Enter elements of Matrix B:\n");
    for (int i = 0; i < r; i++)
        for (int j = 0; j < c; j++)
            scanf("%d", &B[i][j]);

    printf("Enter elements of Matrix C:\n");
    for (int i = 0; i < r; i++)
        for (int j = 0; j < c; j++)
            scanf("%d", &C[i][j]);
    for (int i = 0; i < r; i++)
        for (int j = 0; j < c; j++)
            SUM[i][j] = A[i][j] + B[i][j] + C[i][j];

    printf("Sum of three matrices:\n");
    for (int i = 0; i < r; i++) {
        for (int j = 0; j < c; j++)
            printf("%d ", SUM[i][j]);
        printf("\n");
    }

    return 0;
}
